65  if (reference_file < 0)

71  if (group_header < 0) {

91  H5Fclose(reference_file);
92  throw DCException(getExceptionString(
93  std::string("Failed to read attributes from reference file ") +
94  std::string(filename)));
95  }

132  group_header, mpiPosition.getPointer());
